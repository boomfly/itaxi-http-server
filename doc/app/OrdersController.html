def index
  company_id = current_user.company_id

  conditions = { company_id: company_id }

  case params[:status]
    when 'new'
      conditions[:status] = 'new'
    when 'work'
      conditions[:status] = ['drive_to_client', 'waiting_client', 'on_the_route']
    else
      conditions[:status] = 'new'
  end

  @orders = Order.where(conditions).includes(:users, :stops, :events)

  render json: @orders, :include => [:users, :events, :stops]
end
